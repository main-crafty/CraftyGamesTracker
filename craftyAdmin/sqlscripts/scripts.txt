CREATE TABLE `games` (
  `game_id` int AUTO_INCREMENT NOT NULL,
  `game_name` VARCHAR(255) COLLATE utf8mb4_general_ci UNIQUE NOT NULL,
  `game_description` VARCHAR(255) COLLATE utf8mb4_general_ci NOT NULL,
  `deleted` tinyint(1) NOT NULL,
  primary key (game_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;


CREATE TABLE `users` (
        `userID` int AUTO_INCREMENT NOT NULL,
        `username` varchar(100) COLLATE utf8mb4_general_ci UNIQUE NOT NULL,
        `nickname` varchar(100) COLLATE utf8mb4_general_ci NOT NULL,
        `tiktok` tinyint(1) NOT NULL,
        `deleted` tinyint(1) NOT NULL,
        primary key (userID)
      ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE `userXgames` (
  `userXgameID` int AUTO_INCREMENT NOT NULL,
  `userID` int NOT NULL,
  `gameID` int NOT NULL,
  `deleted` tinyint(1) NOT NULL,
  primary key (userXgameID),
  foreign key (userID) references users(userID),
  foreign key (gameID) references games(game_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
